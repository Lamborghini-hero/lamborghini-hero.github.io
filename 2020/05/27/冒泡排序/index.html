冒泡排序

发表于 2020-05-27

时间复杂度O(n^2),额外空间复杂度(O(1))

算法思路

从数组中第一个数开始比较相邻两个数的大小
将较大的数放在后面的位置
经过一次遍历，会将数组中最大的值放在数组元素的最后一个位置
之后比较前 0 ~ n-1, 0 ~ n-2 ……… 0 个数

public class BubbleSort {

public static void Bubble_sort(int[] arr){
    if(arr.length<2){
        return;
    }

    // arr[i] 为最后一个数
    for(int i = arr.length-1;i>0;i--){
        // 从第一个数开始比较
        for(int j = 0;j<i;j++){
            if(arr[j]>arr[j+1]){
                swap(arr,j,j+1);
            }
        }
    };
}

// 交换arr[i],arr[j]
public static void swap(int[] arr, int i, int j) {
arr[i] = arr[i] ^ arr[j];
arr[j] = arr[i] ^ arr[j];
arr[i] = arr[i] ^ arr[j];
}

// 输出arr
public static void printArray(int[] arr) {
if (arr == null) {
return;
}
for (int i = 0; i < arr.length; i++) {
System.out.print(arr[i] + “ “);
}
System.out.println();
}

public static void main(String[] args) {
    int[] arr = new int[]{34,13, 43 ,23,34,56,43,12};
    Bubble_sort(arr);
    printArray(arr);
}

}
